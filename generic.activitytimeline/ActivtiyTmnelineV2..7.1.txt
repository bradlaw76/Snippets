            records.push(...(res.entities || []));
            while (res.nextLink && records.length < cap) {
                const nextRel = res.nextLink.includes('?') ? '?' + res.nextLink.split('?')[1] : res.nextLink;
                res = await Xrm.WebApi.retrieveMultipleRecords(entity, nextRel);
                records.push(...(res.entities || []));
            }
            return records;
        } catch(e) { return []; }
    };
